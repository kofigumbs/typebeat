The script I used to generate the 808 bits of the SOUL patch:

```rb
bd = []
sd = []
perc = []
rest = []

# get samples by filename in reverse alphanumeric order
`find dsp/samples -name '*.WAV' -exec basename {} \\; | sort`.lines.each do |file|
  case file
  when /BD/
    bd.prepend file
  when /SD/
    sd.prepend file
  when /CB/, /CH/, /CL/, /CP/, /MA/, /RS/
    perc.prepend file
  else
    rest.prepend file
  end
end

# reorder as many samples as possible by taking slices where layout matches:
#     XX XX XX XX XX
#     SD SD SD SD SD
#     BD BD BD BD BD
ordered = []
ordered += perc.reverse # start with the 6 percussive samples
while rest.count > 0 do
  ordered += bd.pop(5).reverse
  ordered += sd.pop(5).reverse
  ordered += rest.pop(5).reverse
end

ordered.each.with_index do |sample, i|
  puts "#{(15 - perc.count + i).to_s.rjust(3, '0')}\t#{sample}"
end
```
